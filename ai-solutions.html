# LeadSite AI — Lead Qualification Agent
def qualify_lead(contact):
  agent = LeadSiteAI(
    model="claude-sonnet",
    tools=["crm_lookup", "score"]
  )
  result = agent.run(contact)
  return {
    "score": result.score,
    "qualified": result.score > 0.75,
    "next": "book_call"
  }

# Output: { score: 0.91, qualified: true, next: "book_call" }
